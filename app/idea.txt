PART 1 – Node.js Backend (AI Layer)

You must regenerate your key. The old one is compromised.

3. server.js
import express from "express";
import axios from "axios";
import cors from "cors";
import dotenv from "dotenv";

dotenv.config();

const app = express();
app.use(cors());
app.use(express.json());

const GROQ_URL = "https://api.groq.com/openai/v1/chat/completions";

async function callGroq(messages) {
  const response = await axios.post(
    GROQ_URL,
    {
      model: "llama-3.1-8b-instant",
      messages,
      temperature: 0.4
    },
    {
      headers: {
        Authorization: `Bearer ${process.env.GROQ_API_KEY}`,
        "Content-Type": "application/json"
      }
    }
  );

  return response.data.choices[0].message.content;
}

4. API 1 – Natural Language → Structured Task
app.post("/ai/create-task", async (req, res) => {
  const { input } = req.body;

  const messages = [
    { role: "system", content: "You are a task management assistant. Convert user text into structured JSON." },
    {
      role: "user",
      content: `Convert this into JSON with title, description, dueDate, priority: ${input}`
    }
  ];

  try {
    const result = await callGroq(messages);
    res.json({ result });
  } catch (err) {
    res.status(500).json({ error: "AI failed" });
  }
});

PART 2 – Expo App Integration

Assume your backend IP is: 192.168.1.5 (use ipconfig to check)

A. Natural Language Task Creation
const createTaskFromText = async (text) => {
  const res = await fetch("http://192.168.1.5:5000/ai/create-task", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ input: text })
  });

  const data = await res.json();
  return data.result;
};


Usage:

const aiTask = await createTaskFromText("Remind me to submit assignment on Friday evening");

PART 4 – Efficiency Rules (Very Important)

Call AI only on button press, never on every keystroke

Do not send unnecessary fields (IDs, timestamps, etc.)

Cache description/priority once generated

Keep prompts short and direct

Temperature 0.4 is ideal (already set)
